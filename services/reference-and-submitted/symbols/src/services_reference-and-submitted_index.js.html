  3 /**
  4  * Handles elements
  5  * @namespace ReferenceAndSubmittedService
  6  */
  7 const getQuestionnaire = require('../../questionnaire/questionnaire-service');
  8 
  9 function createReferenceService() {
 10     /**
 11      * Filter questionnaire object for sectionId of p-applicant-fatal-claim.
 12      * Returns this object or null. Could be extended to make more generic.
 13      *
 14      * @param {object} questionnaire
 15      * @returns {object} questionnaire or null
 16      */
 17     function questionnaireCustomFilter(questionnaire) {
 18         if (
 19             Object.prototype.hasOwnProperty.call(questionnaire, 'sectionId') &&
 20             questionnaire['sectionId'] === 'p-applicant-fatal-claim'
 21         ) {
 22             return questionnaire;
 23         }
 24 
 25         for (let i = 0; i < Object.keys(questionnaire).length; i += 1) {
 26             if (typeof questionnaire[Object.keys(questionnaire)[i]] === 'object') {
 27                 const o = questionnaireCustomFilter(questionnaire[Object.keys(questionnaire)[i]]);
 28                 if (o != null) {
 29                     return o;
 30                 }
 31             }
 32         }
 33 
 34         return null;
 35     }
 36 
 37     /**
 38      * Function extends questionnaireCustomFilter to check if fatal claim and return boolean.
 39      * @param {object} questionnaire
 40      * @returns {boolean} fatalClaim
 41      */
 42 
 43     function checkClaimType(questionnaire) {
 44         const sectionIdSearchFor = questionnaireCustomFilter(questionnaire);
 45         const fatalClaim = sectionIdSearchFor.value;
 46 
 47         return fatalClaim;
 48         // if (typeof fatalClaim !== 'undefined') {
 49         //     return fatalClaim ? claimTypeDefinitions.fatalClaim : claimTypeDefinitions.PIorPOA;
 50         // }
 51         // return null;
 52         // above code needed when outputing claim No. to db.
 53     }
 54 
 55     /**
 56      * Function to be called to get db to add submitted and reference to questionnaire table.
 57      * currently just console logs claim type.
 58      * @param {UUID} questionnaireId
 59      */
 60     function addSubmittedAndReference(questionnaireId) {
 61         // get questionare object from db.
 62         let questionnaire = getQuestionnaire(questionnaireId);
 63 
 64         let typeOfClaim = checkClaimType(questionnaire);
 65 
 66         // print out claim Type.
 67         console.log(typeOfClaim);
 68     }
 69 
 70     return Object.freeze({
 71         questionnaireCustomFilter,
 72         checkClaimType,
 73         addSubmittedAndReference
 74     });
 75 }
 76 
 77 module.exports = createReferenceService;
